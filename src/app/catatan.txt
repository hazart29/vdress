CREATE TABLE userResource (
    id integer PRIMARY KEY,
    uid integer REFERENCES users(uid),
    Chic_Coins integer,
    Glamour_Gems integer,
    Glamour_Dust integer,
    Fashion_Tokens integer,
    Shimmering_Essence integer,
    Glimmering_Essence integer
    

    -- Add comments separately after column definitions
    COMMENT ON COLUMN userResource.Glamour_Gems IS 'gacha gems',
    COMMENT ON COLUMN userResource.Glamour_Dust IS 'gacha exchange item',
    COMMENT ON COLUMN userResource.Chic_Coins IS 'in-game paid currency',
    COMMENT ON COLUMN userResource.Fashion_Tokens IS 'exchange duplicate item' 
    COMMENT ON COLUMN userResource.Shimmering_Essence IS 'limited gacha item' 
    COMMENT ON COLUMN userResource.Glimmering_Essence IS 'standard gacha item' 
);
